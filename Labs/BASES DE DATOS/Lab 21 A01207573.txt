¿Para qué sirve la instrucción GO? 
GO señala el fin de una serie de instrucciones en SQL Server. Además, se puede especificar un count GO[count] para 
especificar el número de veces que se va a correr esta serie de sentencia

recibe: @uclave NUMERIC(5,0),
        @udescripcion VARCHAR(50),
        @ucosto NUMERIC(8,2),
        @uimpuesto NUMERIC(6,2)
y inserta los valores en la tabla de Materiales

modificaMaterial que permite modificar un material que reciba como parámetros las columnas de la tabla materiales y actualice las columnas correspondientes 
con los valores recibidos, para el registro cuya llave sea la clave que se recibe como parámetro. 

CREATE PROCEDURE modificaMaterial
    @uclave NUMERIC(5,0),
    @udescripcion VARCHAR(50),
    @ucosto NUMERIC(8,2),
    @uimpuesto NUMERIC(6,2)
AS
    UPDATE Materiales
	SET Descripcion = @udescripcion,Costo = @ucosto, PorcentajeImpuesto=@uimpuesto
	WHERE Clave=@uclave
GO

eliminaMaterial que elimina el registro de la tabla materiales cuya llave sea la clave que se recibe como parámetro.7
CREATE PROCEDURE deleteMaterial
    @uclave NUMERIC(5,0)
AS
    DELETE FROM Materiales
	WHERE Clave = @uclave
GO 

Desarrollar los procedimientos (almacenados) creaProyecto , modificaproyecto y eliminaproyecto, hacer lo mismo para 
las tablas proveedores y entregan.

CREATE PROCEDURE creaProyecto
    @num NUMERIC(5,0),
    @denom VARCHAR(50)
AS
	INSERT INTO Proyectos VALUES(@num,@denom)
GO

CREATE PROCEDURE modifyProyecto
    @num NUMERIC(5,0),
    @denom VARCHAR(50)
AS
	UPDATE Proyectos
	set Denominacion = @denom
	WHERE Numero = @num
GO

CREATE PROCEDURE deleteProyecto
    @num NUMERIC(5,0)
AS
	DELETE FROM Proyectos
	WHERE Numero = @num
GO


CREATE PROCEDURE creaProveedor
    @rfc VARCHAR(13),
    @razonsocial VARCHAR(50)
AS
	INSERT INTO Proveedores VALUES(@rfc,@razonsocial)
GO

CREATE PROCEDURE modifyProveedor
    @rfc VARCHAR(13),
    @razonsocial VARCHAR(50)
AS
	UPDATE Proveedores
	set RazonSocial = @razonsocial
	WHERE RFC = @rfc
GO

CREATE PROCEDURE deleteProveedor
    @rfc VARCHAR(13)
AS
	DELETE FROM Proveedores
	WHERE RFC = @rfc
GO


CREATE PROCEDURE creaEntrega
    @clave NUMERIC(5),
	@rfc VARCHAR(13),
	@num NUMERIC(5),
	@fecha DATETIME,
	@cantidad NUMERIC(5,2)

AS
	INSERT INTO Entregan VALUES(@clave,@rfc,@num,@fecha,@cantidad)
GO

CREATE PROCEDURE modifyEntrega
	@clave NUMERIC(5),
	@rfc VARCHAR(13),
	@num NUMERIC(5),
	@fecha DATETIME,
	@cantidad NUMERIC(5,2)
AS
	UPDATE Entregan
	set Fecha = @fecha, Cantidad = @cantidad
	WHERE Clave = @clave AND RFC = @rfc AND Numero = @num
GO

CREATE PROCEDURE deleteEntrega
    @clave NUMERIC(5),
	@rfc VARCHAR(13),
	@num NUMERIC(5)
AS
	DELETE FROM Entregan
	WHERE Clave = @clave AND RFC = @rfc AND Numero = @num 
GO
